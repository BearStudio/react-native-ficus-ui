# Modal

Component to render a modal imported from `react-native-modal`

https://github.com/react-native-modal/react-native-modal

## Import

```js
import { Modal } from "react-native-ficus-ui";
```

## Usage

<CodeEditor code={`const TestModal = () => {
  const { isOpen, onOpen, onClose } = useDisclosure();

  return (
    <Box>
      <Button
        onPress={() => {
          onOpen();
        }}
      >
        Show Modal
      </Button>

      <Modal isOpen={isOpen}>
        <Flex bg="white">
          <Button
            h={35}
            w={35}
            position="absolute"
            top={50}
            right={15}
            px={0}
            borderRadius="full"
            colorScheme="gray"
            onPress={() => {
              onClose();
            }}
          >
            <Icon color="white" name="close" />
          </Button>
        </Flex>
      </Modal>
    </Box>
  );
}
render(<TestModal />)`} noInline />
